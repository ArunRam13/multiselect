Examples

var options = {
	startUp: function( $left, $right ){			
		$right.find('option').each(function(index, option){
			$left.find('option[value="' + option.value + '"]').remove();
		});
	},
	sort: false,
	right: '#to2',
	rightAll: '#rightAll2',
	rightSelected: '#rightSelected2',
	leftAll: '#leftAll2',
	leftSelected: '#leftSelected2'
};

$('#from').multiselect(options);
